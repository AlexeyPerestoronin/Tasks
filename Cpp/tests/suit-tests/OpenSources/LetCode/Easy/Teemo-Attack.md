```cpp
class Solution {
    public:
    int findPoisonedDuration(std::vector<int>& timeSeries, int duration) {
      //implement solution here
      int result = duration;
      for (int i = 0; i < timeSeries.size() - 1; ++i) {
        if (timeSeries[i] + duration > timeSeries[i + 1])
          result += timeSeries[i + 1] - timeSeries[i];
        else
          result += duration;
      }
      return result;
    }
  };
```
