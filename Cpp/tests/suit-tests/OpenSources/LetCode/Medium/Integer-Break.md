## Solution

```cpp
class Solution {
  public:
  int integerBreak(int n) 
    {
        //basic cases
        if(n == 2)
            return 1;
        if(n == 3)
            return 2;
        
        int prod = 1;
        
        //factorizing with smaller numbers will give a higher multiplied value
        //we omit the 1s because they don't modify the final solution
        while(n > 0)
        {
            if(n == 4)
            {
                prod *= 4;
                n -= 4;
            }
            else if(n == 2)
            {
                prod *= 2;
                n -= 2;
            }
            else
            {
                prod *= 3;
                n -= 3;
            }
        }
        return prod;
    }
};
```
