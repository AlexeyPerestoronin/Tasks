## Solution
```cpp
  class Solution {
    public:
    uint64_t myAtoi(const char* i_str) {
      uint64_t result{};
      for (int i{ 0 }; i_str[i] != '\0'; ++i) {
        result *= 10;
        result += int(i_str[i] - '0');
      }
      return result;
    }
  };
```
